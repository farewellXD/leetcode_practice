class Solution:
    def calPoints(self, ops):
        """
        :type ops: List[str]
        :rtype: int
        """
        score = []
        
        for op in ops:
            if op =='+':
                score.append(score[-1] + score[-2])
            elif op == 'D':
                score.append(score[-1]*2)
            elif op == 'C':
                score.pop()
            else:
                score.append(int(op))
                
        return sum(score)
